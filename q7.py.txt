class Car:
    def __init__(self, make, model, year):
        if not isinstance(year, int):
            raise ValueError("Year must be an integer")
        self.make = make
        self.model = model
        self.year = year

    def describe_car(self):
        print(f"{self.year} {self.make} {self.model}")
        
my_car = Car("Toyota", "Corolla", 2020)
my_car.describe_car()

2020 Toyota Corolla
